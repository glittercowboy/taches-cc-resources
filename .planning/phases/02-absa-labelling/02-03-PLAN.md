    ```r
    unnest_absa_results <- function(df_absa) {
      # Filter to records with valid ABSA results
      df_valid <- dplyr::filter(df_absa, !absa_parse_error, !absa_api_error,
                                 !sapply(absa_result, is.null))

      # For each record, expand aspects into rows
      df_valid |>
        dplyr::mutate(aspect_data = purrr::map(absa_result, ~ {
          aspects <- .x$aspects
          if (length(aspects) == 0) return(tibble::tibble(
            aspect_id = NA_character_, aspect_label = NA_character_,
            is_novel = FALSE, sentiment = NA_character_,
            evidence = NA_character_, confidence = NA_character_
          ))
          tibble::tibble(
            aspect_id    = sapply(aspects, `[[`, "aspect_id"),
            aspect_label = sapply(aspects, `[[`, "aspect_label"),
            is_novel     = sapply(aspects, function(a) isTRUE(a$is_novel)),
            sentiment    = sapply(aspects, `[[`, "sentiment"),
            evidence     = sapply(aspects, `[[`, "evidence"),
            confidence   = sapply(aspects, `[[`, "confidence")
          )
        })) |>
        tidyr::unnest(aspect_data) |>
        dplyr::select(-absa_result, -absa_parse_error, -absa_api_error)
    }
    ```

    This produces a tidy tibble. Records with no aspects assigned get one row with NA aspect fields (preserving the record in the dataset).

    **2. `validate_sentiment_values(df_tidy)`**: Check all `sentiment` values are in `c("positive","negative","neutral","mixed",NA)`. Log any invalid values.
