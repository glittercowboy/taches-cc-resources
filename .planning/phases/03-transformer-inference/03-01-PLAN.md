    ```python
    def run_batch_inference(df, sentiment_pipe, emotion_pipe, batch_size=32, checkpoint_file=None):
        # Load checkpoint
        processed_ids = set()
        if checkpoint_file and os.path.exists(checkpoint_file):
            ckpt = pd.read_csv(checkpoint_file)
            processed_ids = set(ckpt['id'].astype(str))
            print(f"Resuming: {len(processed_ids)} records already processed")

        remaining = df[~df['id'].astype(str).isin(processed_ids)]
        results = []

        for i in tqdm(range(0, len(remaining), batch_size), desc="Inference"):
            batch = remaining.iloc[i:i+batch_size]
            texts = batch['text'].fillna("").tolist()

            sent_outputs  = sentiment_pipe(texts, truncation=True, max_length=512)
            emot_outputs  = emotion_pipe(texts,  truncation=True, max_length=512)

            for j, row in enumerate(batch.itertuples()):
                result = {"id": str(row.id)}
                # Sentiment: flatten to columns sentiment_negative, sentiment_neutral, sentiment_positive
                for s in sent_outputs[j]:
                    result[f"sentiment_{s['label'].lower()}"] = round(s['score'], 4)
                result["sentiment_label"] = max(sent_outputs[j], key=lambda x: x['score'])['label'].lower()
                # Emotion: flatten to emotion_{label} columns
                for e in emot_outputs[j]:
                    result[f"emotion_{e['label'].lower().replace(' ','_')}"] = round(e['score'], 4)
                result["emotion_label"] = max(emot_outputs[j], key=lambda x: x['score'])['label'].lower()
                results.append(result)

            # Checkpoint every 500 records
            if checkpoint_file and (i + batch_size) % 500 < batch_size:
                pd.DataFrame(results).to_csv(checkpoint_file, index=False)

        return pd.DataFrame(results)
    ```

    **Main block**: load input CSV, run both pipelines, save to parquet with `pyarrow`. Print summary stats (label distribution for sentiment and top 5 emotion labels).

    export_for_inference <- function(df, output_path = here::here("data", "processed", "for_inference.csv")) {
      df_unique <- df |>
        dplyr::distinct(id, .keep_all = TRUE) |>
        dplyr::select(id, text, subreddit, keyword, type)
      readr::write_csv(df_unique, output_path)
      log_msg(paste("Exported", nrow(df_unique), "unique records to", output_path))
      invisible(df_unique)
    }
