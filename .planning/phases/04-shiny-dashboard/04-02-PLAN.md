    library(bslib)
    library(DT)
    library(plotly)
    library(dplyr)
    library(ggplot2)
    library(shinyWidgets)
    library(here)

    # Load pipeline configuration (for paths)
    source(here::here("R", "utils.R"))
    load_config()

    # Load pre-computed dashboard data
    dashboard_data_path <- here::here("data", "processed", "09_dashboard_data.rds")
    if (!file.exists(dashboard_data_path)) {
      stop("Dashboard data not found. Run scripts/08_prepare_dashboard.R first.")
    }
    dash <- readRDS(dashboard_data_path)

    # Expose top-level objects for convenience
    records      <- dash$records          # Full tidy dataset for reactive filtering
    filter_opts  <- dash$filter_options   # Dropdown options
    meta         <- dash$metadata         # Pipeline metadata

    # App theme
    APP_THEME <- bslib::bs_theme(
      version = 5,
      bootswatch = "flatly",
      primary = "#2c7bb6",
      base_font = bslib::font_google("Inter")
    )

    Create `app/server.R` scaffold with placeholder outputs:
    ```r
    server <- function(input, output, session) {

      # ── Reactive filtered dataset ──────────────────────────────────────
      filtered_records <- shiny::reactive({
        records  # Will be filtered by inputs in Plan 04-03
      })

      # ── Overview value boxes ───────────────────────────────────────────
      output$n_records    <- shiny::renderText({ format(length(unique(filtered_records()$id)), big.mark=",") })
      output$n_subreddits <- shiny::renderText({ length(unique(filtered_records()$subreddit)) })
      output$n_aspects    <- shiny::renderText({ sum(!is.na(filtered_records()$aspect_id)) })
      output$date_range   <- shiny::renderText({
        r <- range(filtered_records()$created_utc, na.rm = TRUE)
        paste(format(r, "%b %Y"), collapse = " – ")
      })

      # Pipeline metadata
      output$pipeline_meta <- shiny::renderUI({
        shiny::tagList(
          shiny::p(paste("Records analysed:", format(meta$n_unique_ids, big.mark=","))),
          shiny::p(paste("Pipeline run:", format(meta$pipeline_run_date, "%Y-%m-%d %H:%M")))
        )
      })

      # Placeholder plot outputs (will be implemented in Plans 04-03 and 04-04)
      output$plot_aspect_dist     <- shiny::renderPlot({ plot(1, main="Coming in Plan 04-04") })
      output$plot_sentiment_aspect <- shiny::renderPlot({ plot(1, main="Coming in Plan 04-04") })
      output$plot_sentiment_time  <- shiny::renderPlot({ plot(1, main="Coming in Plan 04-04") })
      output$plot_emotion_dist    <- shiny::renderPlot({ plot(1, main="Coming in Plan 04-04") })
      output$plot_emotion_aspect  <- shiny::renderPlot({ plot(1, main="Coming in Plan 04-04") })
      output$table_records        <- DT::renderDataTable({ head(filtered_records(), 50) })
    }
    ```
