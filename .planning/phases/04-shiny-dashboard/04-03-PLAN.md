  <name>Task 2: Implement filtered_records() reactive and reset logic in server.R</name>
  <files>app/server.R</files>
  <action>
    Replace the stub `filtered_records <- shiny::reactive({ records })` with full filtering logic:

    ```r
    # ── Reset filter button ────────────────────────────────────────────
    shiny::observeEvent(input$btn_reset, {
      shinyWidgets::updatePickerInput(session, "filter_subreddits", selected = filter_opts$subreddits)
      shinyWidgets::updatePickerInput(session, "filter_keywords",   selected = filter_opts$keywords)
      shinyWidgets::updatePickerInput(session, "filter_aspects",    selected = "all")
      shiny::updateCheckboxGroupInput(session, "filter_sentiment",  selected = c("positive","negative","neutral","mixed"))
      shiny::updateCheckboxGroupInput(session, "filter_type",       selected = c("post","comment"))
      shiny::updateDateRangeInput(session, "filter_dates",
        start = min(records$created_utc, na.rm=TRUE),
        end   = max(records$created_utc, na.rm=TRUE))
    })

    # ── Main reactive filtered dataset ────────────────────────────────
    filtered_records <- shiny::reactive({
      df <- records

      # Date filter
      if (!is.null(input$filter_dates)) {
        df <- dplyr::filter(df,
          as.Date(created_utc) >= input$filter_dates[1],
          as.Date(created_utc) <= input$filter_dates[2]
        )
      }

      # Subreddit
      if (!is.null(input$filter_subreddits) && length(input$filter_subreddits) > 0) {
        df <- dplyr::filter(df, subreddit %in% input$filter_subreddits)
      }

      # Keyword
      if (!is.null(input$filter_keywords) && length(input$filter_keywords) > 0) {
        df <- dplyr::filter(df, keyword %in% input$filter_keywords)
      }

      # Aspect — "all" means no filter; else filter to rows matching that aspect
      if (!is.null(input$filter_aspects) && input$filter_aspects != "all") {
        df <- dplyr::filter(df, aspect_label == input$filter_aspects | is.na(aspect_label))
      }

      # Aspect sentiment (only applies to rows with an aspect)
      if (!is.null(input$filter_sentiment)) {
        df <- dplyr::filter(df,
          is.na(aspect_sentiment) | aspect_sentiment %in% input$filter_sentiment
        )
      }

      # Post type
      if (!is.null(input$filter_type) && length(input$filter_type) > 0) {
        df <- dplyr::filter(df, type %in% input$filter_type)
      }

      df
    })

    # ── Reactive record count for UI feedback ─────────────────────────
    output$filter_count <- shiny::renderText({
      n <- length(unique(filtered_records()$id))
      paste(format(n, big.mark=","), "records match current filters")
    })
    ```
